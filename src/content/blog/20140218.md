<p>like数の取得</p>
<pre class="brush: jscript; title: ; notranslate" title="">&lt;script&gt;
$(function() {
	var url= 'http://yahoo.co.jp'; //任意のURL
	$.getJSON('http://graph.facebook.com/?id='+encodeURI(url)+'&amp;amp;callback=?', {}, function(json) {
		$('#like').text(json.shares);  //任意のセレクタ
	})
});
&lt;/script&gt;
</pre>

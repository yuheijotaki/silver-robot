@for $i from 1 through 100 {
	@if $i &lt; 10 {
		$num: "0#{$i}";
	} @else {
		$num: $i;
	}
	.w_#{$num} { //.w_XX
		width: percentage($i/100); //width: XX%;
	}
	.mr_#{$num} { //.mr_XX
		margin-right: percentage($i/100); //margin-right: XX%;
	}
}
</pre>
